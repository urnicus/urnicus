function sortTable(columnIndex) {
  var table = document.getElementById("stateTable");
  var rows = Array.from(table.querySelectorAll("tbody tr"));
  var isNumeric = columnIndex === 1; // Cost column

  rows.sort(function(a, b) {
    var aValue = a.cells[columnIndex].textContent.trim();
    var bValue = b.cells[columnIndex].textContent.trim();

    if (isNumeric) {
      // Extract numeric values from cost strings
      var aNum = extractNumber(aValue);
      var bNum = extractNumber(bValue);
      return aNum - bNum;
    } else {
      return aValue.localeCompare(bValue);
    }
  });

  var tbody = table.querySelector("tbody");
  rows.forEach(function(row) {
    tbody.appendChild(row);
  });
}

function extractNumber(str) {
  if (str.toLowerCase().includes('free')) return 0;
  if (str.toLowerCase().includes('varies')) return 999999;
  // Extract first number from string (handles $X, ~$X, $X-$Y)
  var match = str.match(/[\d,]+/);
  if (match) {
    return parseInt(match[0].replace(/,/g, ''));
  }
  return 0;
}
